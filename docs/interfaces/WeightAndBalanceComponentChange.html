Interface WeightAndBalanceComponentChange

Used to represent either an initial entry or a single component change. If this is an externally calculated entry then the item weight and arm will be left blank.

interface WeightAndBalanceComponentChange {
    aircraftArm: number;
    aircraftWeight: number;
    itemArm?: number;
    itemWeightChange?: number;
    nonLiftingPartsWeight: number;
    weightChangeInFuselage?: boolean;
}
